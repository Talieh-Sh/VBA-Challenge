Sub YearlyChange()Dim ws As Worksheet'Loop through all worksheetsFor Each ws In Worksheets'Declare last_rowDim last_row As Long'Determine the last rowlast_row = ws.Cells(Rows.Count, 1).End(xlUp).Row'Set an initial variable for holding the ticker nameDim ticker_name As String'Set an initial  variable for opening price at the beginning of the given yearDim open_price As Double'Set an initial variable for closing price at the end of the given yearDim close_price As Double'Keep track of the location for ticker in the summary tableDim summary_table_row As Integersummary_table_row = 2'Set an initial variable for total stock variableDim total_stock_volume_per_ticker As Doubletotal_stock_volume_per_ticker = 0'Set the headingsws.Range("J1").Value = "Ticker"ws.Range("K1").Value = "YearlyChange"ws.Range("L1").Value = "Percent Change"ws.Range("M1").Value = "Total Stock Volume "'Loop through all rowsFor i = 1 To last_row   'for all rows except i=1 we have to stack over total_stock_volume_per_ticket and when the ticker changes we have to sho the result of suumation in the result spot    If i <> 1 Then        total_stock_volume_per_ticker = total_stock_volume_per_ticker + (ws.Cells(i, 7).Value)    End If         'check if we are still within the same ticker, if it is not...    If ws.Cells(i + 1, 1).Value <> ws.Cells(i, 1).Value Then                           If i <> 1 Then         closing_price = ws.Cells(i, 6).Value          ws.Range("K" & summary_table_row).Value = closing_price - opening_price                    If ws.Range("K" & summary_table_row).Value > 0 Then            ws.Range("K" & summary_table_row).Interior.ColorIndex = 4          Else            ws.Range("K" & summary_table_row).Interior.ColorIndex = 3          End If                    ws.Range("L" & summary_table_row).Value = (closing_price - opening_price) / (opening_price)          ws.Range("L" & summary_table_row).NumberFormat = "0.00%"                    If ws.Range("L" & summary_table_row).Value > 0 Then            ws.Range("L" & summary_table_row).Interior.ColorIndex = 4          Else            ws.Range("L" & summary_table_row).Interior.ColorIndex = 3          End If                    ws.Range("M" & summary_table_row).Value = total_stock_volume_per_ticker          total_stock_volume_per_ticker = 0          summary_table_row = summary_table_row + 1        End If                 If i <> last_row Then             ticker_name = ws.Cells(i + 1, 1).Value             ws.Range("J" & summary_table_row).Value = ticker_name             opening_price = ws.Cells(i + 1, 3).Value        End If         End If  Next i'Bonus:ws.Range("P2").Value = "Greatest % increase"ws.Range("P3").Value = "Greatest % Decrease"ws.Range("P4").Value = "Greatest Total Volume"ws.Range("Q1").Value = "Ticker"ws.Range("r1").Value = "Value"'Calculating the BonusDim greatest_increase_ticker As StringDim greatest_increase_value As DoubleDim greatest_decrease_ticker As StringDim greatest_decrease_value As DoubleDim greates_total_valume_ticker As StringDim greates_total_valume_value As DoubleDim k As LongDim last_row_2 As Long'Determine the last row of the Summary partlast_row_2 = ws.Cells(Rows.Count, 10).End(xlUp).Rowgreatest_increase_ticker_value = 0greatest_decrease_value = 0greates_total_valume_value = 0'for loop over rows in the summary section to find the variablesFor k = 2 To last_row_2    If Range("L" & k).Value > greatest_increase_ticker_value Then        greatest_increase_ticker_value = ws.Range("L" & k).Value        greatest_increase_ticker = ws.Range("J" & k).Value    End If        If Range("L" & k).Value < greatest_decrease_value Then        greatest_decrease_value = ws.Range("L" & k).Value        greatest_decrease_ticker = ws.Range("J" & k).Value    End If        If Range("M" & k).Value > greates_total_valume_value Then        greates_total_valume_value = ws.Range("M" & k).Value        greates_total_valume_ticker = ws.Range("J" & k).Value    End If    Next kws.Range("Q2").Value = greatest_increase_tickerws.Range("R2").Value = greatest_increase_ticker_valuews.Range("R2").NumberFormat = "0.00%"ws.Range("Q3").Value = greatest_decrease_tickerws.Range("R3").Value = greatest_decrease_valuews.Range("Q4").Value = greates_total_valume_tickerws.Range("R4").Value = greates_total_valume_value'Autofit column J to Nws.Columns("J:R").AutoFitNext wsEnd Sub
